266  template<typename... Args> int handle_tperrno ( int _tperrno, const char *msg, const Args&... args) {
267 
268  switch ( _tperrno ) {
269  case TPEINVAL:
270  case TPEPERM:
271  case TPENOENT:
272  case TPEITYPE:
273  case TPEOTYPE:
274  case TPETRAN:
275  case TPEPROTO:
276  case TPESYSTEM:
277  case TPEOS:
278  case TPELIMIT:
279  {
280  throw tuxedo_exception ( _tperrno, msg, args... );
281  }
282  break;
283  case TPEBLOCK:
284  {
285  throw blocking_exception ( msg, args... );
286  }
287  break;
288  case TPGOTSIG:
289  {
290  throw interrupt_exception ( msg, args... );
291  }
292  break;
293  case TPESVCERR:
294  {
295  throw service_exception ( msg, args... );
296  }
297  break;
298  case TPETIME:
299  {
300  throw timeout_exception ( msg, args... );
301  }
302  break;
303  case TPESVCFAIL:
304  // return application specific error number instead
305  // as the application will probably know what to do
306  _tperrno = ( tpurcode > 0 ? tpurcode : -1 );
307  break;
308 
309  default:
310  throw tuxedo_exception (_tperrno, catgets ( _catd, CATD_ATMI_SET, 33,"Never heard about this tperrno (%d)."), _tperrno );
311  };
312 
313  return _tperrno;
314  };
