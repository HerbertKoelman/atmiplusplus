233  template<typename... Args> int handle_tperrno ( int _tperrno, const char *msg, const Args&... args) {
234 
235  switch ( _tperrno ) {
236  case TPEINVAL:
237  case TPEPERM:
238  case TPENOENT:
239  case TPEITYPE:
240  case TPEOTYPE:
241  case TPETRAN:
242  case TPEPROTO:
243  case TPESYSTEM:
244  case TPEOS:
245  case TPELIMIT:
246  {
247  throw tuxedo_exception ( _tperrno, msg, args... );
248  }
249  break;
250  case TPEBLOCK:
251  {
252  throw blocking_exception ( msg, args... );
253  }
254  break;
255  case TPGOTSIG:
256  {
257  throw interrupt_exception ( msg, args... );
258  }
259  break;
260  case TPESVCERR:
261  {
262  throw service_exception ( msg, args... );
263  }
264  break;
265  case TPETIME:
266  {
267  throw timeout_exception ( msg, args... );
268  }
269  break;
270  case TPESVCFAIL:
271  // return application specific error number instead
272  // as the application will probably know what to do
273  _tperrno = ( tpurcode > 0 ? tpurcode : -1 );
274  break;
275 
276  default:
277  throw tuxedo_exception (_tperrno, catgets ( _catd, CATD_ATMI_SET, 33,"Never heard about this tperrno (%d)."), _tperrno );
278  };
279 
280  return _tperrno;
281  };
