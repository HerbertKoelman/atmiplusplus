268  template<typename... Args> int handle_tperrno ( int _tperrno, const char *msg, const Args&... args) {
269 
270  switch ( _tperrno ) {
271  case TPEINVAL:
272  case TPEPERM:
273  case TPENOENT:
274  case TPEITYPE:
275  case TPEOTYPE:
276  case TPETRAN:
277  case TPEPROTO:
278  case TPESYSTEM:
279  case TPEOS:
280  case TPELIMIT:
281  {
282  throw tuxedo_exception ( _tperrno, msg, args... );
283  }
284  break;
285  case TPEBLOCK:
286  {
287  throw blocking_exception ( msg, args... );
288  }
289  break;
290  case TPGOTSIG:
291  {
292  throw interrupt_exception ( msg, args... );
293  }
294  break;
295  case TPESVCERR:
296  {
297  throw service_exception ( msg, args... );
298  }
299  break;
300  case TPETIME:
301  {
302  throw timeout_exception ( msg, args... );
303  }
304  break;
305  case TPESVCFAIL:
306  // return application specific error number instead
307  // as the application will probably know what to do
308  _tperrno = ( tpurcode > 0 ? tpurcode : -1 );
309  break;
310 
311  default:
312  throw tuxedo_exception (_tperrno, catgets ( _catd, CATD_ATMI_SET, 33,"Never heard about this tperrno (%d)."), _tperrno );
313  };
314 
315  return _tperrno;
316  };
