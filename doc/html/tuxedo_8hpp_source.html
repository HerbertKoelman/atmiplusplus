269  template<typename... Args> int handle_tperrno ( int _tperrno, const char *msg, const Args&... args) {
270 
271  switch ( _tperrno ) {
272  case TPEINVAL:
273  case TPEPERM:
274  case TPENOENT:
275  case TPEITYPE:
276  case TPEOTYPE:
277  case TPETRAN:
278  case TPEPROTO:
279  case TPESYSTEM:
280  case TPEOS:
281  case TPELIMIT:
282  {
283  throw tuxedo_exception ( _tperrno, msg, args... );
284  }
285  break;
286  case TPEBLOCK:
287  {
288  throw blocking_exception ( msg, args... );
289  }
290  break;
291  case TPGOTSIG:
292  {
293  throw interrupt_exception ( msg, args... );
294  }
295  break;
296  case TPESVCERR:
297  {
298  throw service_exception ( msg, args... );
299  }
300  break;
301  case TPETIME:
302  {
303  throw timeout_exception ( msg, args... );
304  }
305  break;
306  case TPESVCFAIL:
307  // return application specific error number instead
308  // as the application will probably know what to do
309  _tperrno = ( tpurcode > 0 ? tpurcode : -1 );
310  break;
311 
312  default:
313  throw tuxedo_exception (_tperrno, catgets ( _catd, CATD_ATMI_SET, 33,"Never heard about this tperrno (%d)."), _tperrno );
314  };
315

317  };
