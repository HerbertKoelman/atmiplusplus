274  template<typename... Args> int handle_tperrno ( int _tperrno, const char *msg, const Args&... args) {
275 
276  switch ( _tperrno ) {
277  case TPEINVAL:
278  case TPEPERM:
279  case TPENOENT:
280  case TPEITYPE:
281  case TPEOTYPE:
282  case TPETRAN:
283  case TPEPROTO:
284  case TPESYSTEM:
285  case TPEOS:
286  case TPELIMIT:
287  {
288  throw tuxedo_exception ( _tperrno, msg, args... );
289  }
290  break;
291  case TPEBLOCK:
292  {
293  throw blocking_exception ( msg, args... );
294  }
295  break;
296  case TPGOTSIG:
297  {
298  throw interrupt_exception ( msg, args... );
299  }
300  break;
301  case TPESVCERR:
302  {
303  throw service_exception ( msg, args... );
304  }
305  break;
306  case TPETIME:
307  {
308  throw timeout_exception ( msg, args... );
309  }
310  break;
311  case TPESVCFAIL:
312  // return application specific error number instead
313  // as the application will probably know what to do
314  _tperrno = ( tpurcode > 0 ? tpurcode : -1 );
315  break;
316 
317  default:
318  throw tuxedo_exception (_tperrno, catgets ( _catd, CATD_ATMI_SET, 33,"Never heard about this tperrno (%d)."), _tperrno );
319  };
320 
321  return _tperrno;
322  };
